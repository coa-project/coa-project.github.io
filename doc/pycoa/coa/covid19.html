553   def get_posteriors(self,sr, window=7, min_periods=1):
554        # We create an array for every possible value of Rt
555        R_T_MAX = 12
556        r_t_range = np.linspace(0, R_T_MAX, R_T_MAX*100+1)
557
558        # Gamma is 1/serial interval
559        # https://wwwnc.cdc.gov/eid/article/26/6/20-0357_article
560        GAMMA = 1/7
561
562        lam = sr[:-1].values * np.exp(GAMMA * (r_t_range[:, None] - 1))
563
564        # Note: if you want to have a Uniform prior you can use the following line instead.
565        # I chose the gamma distribution because of our prior knowledge of the likely value
566        # of R_t.
567
568        # prior0 = np.full(len(r_t_range), np.log(1/len(r_t_range)))
569        prior0 = np.log(sps.gamma(a=3).pdf(r_t_range) + 1e-14)
570
571        likelihoods = pd.DataFrame(
572            # Short-hand way of concatenating the prior and likelihoods
573            data = np.c_[prior0, sps.poisson.logpmf(sr[1:].values, lam)],
574            index = r_t_range,
575            columns = sr.index)
576
577        # Perform a rolling sum of log likelihoods. This is the equivalent
578        # of multiplying the original distributions. Exponentiate to move
579        # out of log.
580        posteriors = likelihoods.rolling(window,
581                                     axis=1,
582                                     min_periods=min_periods).sum()
583        posteriors = np.exp(posteriors)
584
585        # Normalize to 1.0
586        posteriors = posteriors.div(posteriors.sum(axis=0), axis=1)
587
588        return posteriors

554   def get_posteriors(self,sr, window=7, min_periods=1):
555        # We create an array for every possible value of Rt
556        R_T_MAX = 12
557        r_t_range = np.linspace(0, R_T_MAX, R_T_MAX*100+1)
558
559        # Gamma is 1/serial interval
560        # https://wwwnc.cdc.gov/eid/article/26/6/20-0357_article
561        GAMMA = 1/7
562
563        lam = sr[:-1].values * np.exp(GAMMA * (r_t_range[:, None] - 1))
564
565        # Note: if you want to have a Uniform prior you can use the following line instead.
566        # I chose the gamma distribution because of our prior knowledge of the likely value
567        # of R_t.
568
569        # prior0 = np.full(len(r_t_range), np.log(1/len(r_t_range)))
570        prior0 = np.log(sps.gamma(a=3).pdf(r_t_range) + 1e-14)
571
572        likelihoods = pd.DataFrame(
573            # Short-hand way of concatenating the prior and likelihoods
574            data = np.c_[prior0, sps.poisson.logpmf(sr[1:].values, lam)],
575            index = r_t_range,
576            columns = sr.index)
577
578        # Perform a rolling sum of log likelihoods. This is the equivalent
579        # of multiplying the original distributions. Exponentiate to move
580        # out of log.
581        posteriors = likelihoods.rolling(window,
582                                     axis=1,
583                                     min_periods=min_periods).sum()
584        posteriors = np.exp(posteriors)
585
586        # Normalize to 1.0
587        posteriors = posteriors.div(posteriors.sum(axis=0), axis=1)
588
589        return posteriors
